version: '3.4'

services:

  flask:
    container_name: flask
    build:
      context: ./services/flask
      dockerfile: Dockerfile-dev
    volumes:
      - ./services/flask:/usr/src/app
    ports:
      - 5001:5000
    environment:
      - FLASK_DEBUG=1
      - ENVIRONMENT_TYPE=development
      - DATABASE_URL=postgres://postgres:${postgres_password}@flask-db:5432/dev_db
      - DATABASE_TEST_URL=postgres://postgres:${postgres_password}@flask-db:5432/test_db
    depends_on:
      - flask-db
    links:
      - flask-db

  flask-db:
    container_name: flask-db
    build:
      context: ./services/flask/project/db
      dockerfile: Dockerfile
    ports:
      - 5435:5432
    environment:
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=${postgres_password}
    volumes:
      - flask-db_data:/var/lib/postgresql/data

  nginx:
    container_name: nginx
    build:
      context: ./services/nginx
      dockerfile: Dockerfile-dev
    restart: always
    ports:
      - 80:80
    depends_on:
      - flask
      - client

  client:
    container_name: client
    build:
      context: ./services/client
      dockerfile: Dockerfile-dev
    volumes:
      - './services/client:/usr/src/app'
      - '/usr/src/app/node_modules'
    ports:
      - '3007:3000'
    environment:
      - NODE_ENV=development
      - REACT_APP_USERS_SERVICE_URL=http://127.0.0.1
    depends_on:
      - flask
    links:
      - flask

volumes:
  flask-db_data:
    external:
      name: gtaWebApp_data
